## Task 21: AgentLoop 集成全部新模块

**Files:**
- Modify: `src-tauri/src/services/agent_loop.rs`

- [ ] **Step 1: 更新 run_turn 内部集成**

```rust
// 循环开始前
let system = prompt_manager.build_system_prompt(dynamic_context);
let mut data_context = DataContext::new();
let mut context_manager = ContextManager::new(context_config.clone());
let mut last_input_tokens: Option<u32> = None;
let mut counted_entries: usize = 0;

loop {
    let all_messages = history_to_api_messages(&history) 
        + entries_to_messages(&new_entries);
    
    // 第一层裁剪
    let mut trimmed = context_manager.trim_old_tool_results(all_messages);
    
    // Token 估算
    let estimated = context_manager.estimate_total(&new_entries, counted_entries);
    
    // 第二层压缩
    let (final_messages, summary_entry) = if estimated > compact_threshold {
        context_manager.compact_with_llm(trimmed, provider, model).await?
    } else { (trimmed, None) };
    
    // 调用 LLM
    let response = provider.chat_stream(system, final_messages, model, tools).await;
    
    // 更新计数
    last_input_tokens = Some(response.usage.input_tokens);
    counted_entries = new_entries.len();
    
    // 工具执行后更新 DataContext
    if tool_name == "Read" { data_context.record_file_read(...); }
}
```
